pipeline {
	agent any
	stages {
		stage('Checkout') {
			steps {
				git 'https://github.com/CharlotteUppsall/auto21.git'
			}
		}
		stage('Build') {
			steps {
				bat 'mvn compile'
			}
		}
		stage('Test') {
			steps {
				bat 'mvn test'
			}
			post {
				always {
					junit '/*TEST.xml'
				}
			}
		}
	}
}
